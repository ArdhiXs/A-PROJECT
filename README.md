```bash
> pkg update && pkg upgrade
> pkg install bash -y
> pkg install unzip -y
> pkg install git -y
> pkg install nodejs -y
> pkg install ffmpeg -y
> pkg install imagemagick -y
> git clone https://github.com/ardhixsquerpants/A-PROJECT
> cd A-PROJECT
> unzip A-PROJECT.zip
> bash install.sh
> node index.js
```
